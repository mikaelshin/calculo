\paragraph{Ex 1:}
    $\bigint \ln(x)\ dx$\\
     $f' = 1$, \hspace{1em}
     $f = x$, \hspace{1em}
     $g = \ln(x)$, \hspace{1em}
     $g' = \dfrac{1}{x}$
     
    \begin{align*}
    \int \ln(x)dx
     & = \int 1 \cdot \ln(x)dx \\\\
     & = x \cdot \ln x - \int x\cdot\dfrac{1}{x}dx \\\\
     & = x \cdot \ln x - x \\\\
     & = x(\ln x - 1).\\\\\\
    \end{align*}

\paragraph{Ex 2:}
    $\bigint x\cdot\ln(x)\ dx$ \\ 
     $\\ f' = x$, \hspace{1em}
     $f = \dfrac{x^{2}}{2}$, \hspace{1em}
     $g = \ln(x)$, \hspace{1em}
     $g' = \dfrac{1}{x}$
     
    \begin{align*}
    \int x \cdot \ln(x) \ dx
     & = \dfrac{x^{2}}{2} \cdot \ln x - \int \dfrac{x^{2}}{2} \cdot \dfrac{1}{x} \ dx \\\\
     & = \dfrac{x^{2}}{2} \cdot \ln x - \dfrac{1}{2} \cdot \int x \ dx \\\\
     & = \dfrac{x^{2}}{2} \cdot \ln x - \dfrac{1}{2} \cdot \dfrac{x^{2}}{2} \\\\
     & = \dfrac{x^{2}}{2} \cdot \left(\ln x - \dfrac{1}{2}\right).\\\\
    \end{align*}

\paragraph{Ex 3:}
    $\bigint x^{n}\cdot\ln(x)\ dx$\\
     $\\ f' = x^{n},$ \hspace{1em}
     $f = \dfrac{x^{n + 1}}{n + 1}$, \hspace{1em}
     $g = \ln x$, \hspace{1em}
     $g' = \dfrac{1}{x}$
     
    \begin{align*}
    \int x^{n} \cdot \ln(x) \ dx
     & = \dfrac{x^{n+1}}{n+1} \cdot \ln x - \int \dfrac{x^{n+1}}{n+1} \cdot \dfrac{1}{x} \ dx \\\\
     & = \dfrac{x^{n+1}}{n+1} \cdot \ln x - \int \dfrac{x^{n}}{n+1} \ dx \\\\
     & = \dfrac{x^{n+1}}{n+1} \cdot \ln x - \dfrac{1}{n+1} \cdot \int x^{n} \ dx \\\\
     & = \dfrac{x^{n+1}}{n+1} \cdot \ln x - \dfrac{1}{n+1} \cdot \dfrac{x^{n+1}}{n+1} \\\\
     & = \dfrac{x^{n+1}}{n+1} \cdot \left(\ln x - \dfrac{1}{n+1}\right).\\\\
    \end{align*}

\paragraph{Ex 4:}
    $\bigint x \cdot \ln(1 + x)\ dx$ \\
     $\\ f' = x$, \hspace{1em}
     $f = \dfrac{x^{2}}{2}$, \hspace{1em}
     $g = \ln(1 + x)$, \hspace{1em}
     $g' = \dfrac{1}{x + 1}$
    
    
    \begin{align*}
    \int x\cdot \ln(1 + x)\ dx
     & = \dfrac{x^{2}}{2} \cdot \ln(1 + x) - \int\dfrac{x^{2}}{2} \cdot \dfrac{1}{x + 1}\  dx\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 + x) - \dfrac{1}{2}\cdot\int \dfrac{x^{2}}{1 + x}\ dx\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 + x) - \dfrac{1}{2}\cdot \left[\dfrac{(x - 1)^{2}}{2} + \ln(1+x)\right]\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 + x) - \dfrac{(x - 1)^{2}}{4} - \dfrac{1}{2}\ln(1 + x)\\\\
     & =\dfrac{x^{2}-1}{2}\cdot\ln(1+x) - \dfrac{(x - 1)^{2}}{4}.
    \end{align*}
    
    obs: $\bigints \dfrac{x^{2}}{x - 1}$: Referência usada pelo Ex 6.

\paragraph{Ex 5:}
    $\bigint x\cdot \ln(1 - x) \ dx$ \\
     $\\ f' = x$, \hspace{1em}
     $f = \dfrac{x^{2}}{2}$, \hspace{1em}
     $g = \ln(1 - x)$, \hspace{1em}
     $g' = - \dfrac{1}{1 - x}$
    
    \begin{align*}
    \int x\cdot\ln(1 - x)\ dx
     & = \dfrac{x^{2}}{2} \cdot \ln(1 - x) - \int\dfrac{x^{2}}{2} \cdot \left(- \dfrac{1}{x - 1}\right) \ dx\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 - x) + \dfrac{1}{2}\cdot\int \dfrac{x^{2}}{1 - x}\ dx\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 - x) + \dfrac{1}{2}\cdot \left[- \dfrac{(x + 1)^{2}}{2} - \ln(1 - x) \right]\\\\
     & = \dfrac{x^{2}}{2} \cdot \ln(1 - x) - \dfrac{(x + 1)^{2}}{4} - \dfrac{1}{2}\ln(1 - x)\\\\
     & =\dfrac{x^{2}-1}{2}\cdot \ln(1 - x) - \dfrac{(1 + x)^{2}}{4}.
    \end{align*}
    
    
     obs: $\bigints \dfrac{x^{2}}{x - 1}$: Referência usada pelo Ex 7.
    
% Ex 6 -------------------------------------------------------------
\paragraph{Ex 6:}
    $\bigint \dfrac{x^{2}}{1 + x} \ dx$ \\
     $\\\\ u = 1 + x$, \hspace{1em}
     $du = dx$, \hspace{1em}
     $x^{2} = (u - 1)^{2}$, \hspace{1em}
     
    \begin{align*}
    \int \dfrac{x^{2}}{1 + x} \ dx
     & = \int \dfrac{(u-1)^{2}}{u} \ du \\\\
     & = \int \dfrac{u^{2} - 2u + 1}{u} \ du \\\\
     & = \int u + \dfrac{1}{u} - 2 \ du\\\\
     & = \int u \ du + \int \dfrac{1}{u} \ du - 2 \cdot \int 1 \ du \\\\
     & = \dfrac{u^{2}}{2} + \ln u - 2 \cdot u\\\\
     \end{align*}
     \begin{align*}
     & = \dfrac{(1 + x)^{2}}{2} + \ln(1 + x) - 2 \cdot (1 + x)\\\\
     & = \dfrac{1 + 2x + x^{2}}{2} - 2 - 2x + \ln(1 + x)\\\\
     & = \ln(1 + x) + \dfrac{x^{2}}{2} - x + C\\\\
     & = \ln(1 + x) + \dfrac{(x - 1)^{2}}{2}.
    \end{align*}
    
% Ex 7 -------------------------------------------------------------
\paragraph{Ex 7:}
    $\bigint \dfrac{x^{2}}{1 - x} \ dx$\\
     $\\\\ u = 1 - x$, \hspace{1em}
     $du = -dx$, \hspace{1em}
     $x^{2} = (1 - u)^{2}$, \hspace{1em}
     
    \begin{align*}
    \int \dfrac{x^{2}}{1 - x} \ dx
     & = - \int \dfrac{(1 - u)^{2}}{u} \ du \\\\
     & = - \int \dfrac{u^{2} - 2u + 1}{u} \ du \\\\
     & = - \left(\int u \ du - 2 \cdot \int 1 \ du + \int \dfrac{1}{u} \ du \right) \\\\
     & = - \dfrac{u^{2}}{2} - \ln u + 2 \cdot u\\\\
     & = - \ln(1 - x) - \dfrac{(1 - x)^{2}}{2} + 2 \cdot (1 - x)\\\\
     & = - \ln(1 - x) - \left( \dfrac{x^{2} - 2x + 1}{2} \right) + (2 - 2x) \\\\
     & = - \ln(1 - x) - \dfrac{x^{2}}{2} + x - \dfrac{1}{2} + 2 - 2x\\\\
     & = - \ln(1 - x) - \dfrac{x^{2}}{2} - x + C\\\\
     & = - \ln(1 - x) - \dfrac{(x - 1)^{2}}{2}.\\\\
    \end{align*}

    \paragraph{Ex 8:}
    $\bigint \ln(1 + x)\ dx$ \\
     $u = 1 + x$, \hspace{1em}
     $du = dx$, \hspace{1em}
     $f = u$, \hspace{1em}
     $f' = 1$, \hspace{1em}
     $g = \ln(u)$, \hspace{1em}

\paragraph{Ex 9:}
    $\bigint \ln(1 - x)\ dx$ \\
     $u = 1 - x$, \hspace{1em}
     $du = -dx$, \hspace{1em}
     $f = u$, \hspace{1em}
     $f' = 1$, \hspace{1em}
     $g = \ln(u)$, \hspace{1em}
     $g' = \dfrac{1}{u}$
     
    \begin{align*}
    \int \ln(1 - x)\ dx
     & = - \int \ln(u) \ du \\\\
     & = - \int 1 \cdot \ln u \ du\\\\
     & = - \left (u \cdot \ln u - \int u \cdot \dfrac{1}{u} \ du \right) \\\\
     & = - \left( u \cdot \ln u - \int 1 \ du \right) \\\\
     & = -u \cdot \ln u + u \\\\
     & = - (1 + x) \cdot \ln(1 - x) + (1 - x) \\\\
     & = - (1 + x) \cdot \ln(1 - x) - x.
    \end{align*}

\paragraph{Ex 10:}
    $\bigint \ln(1 + x^{2})\ dx$ 
    
     $\\f = x$, \hspace{1em}
     $f' = 1$, \hspace{1em}
     $g = \ln (1 + x^{2})$, \hspace{1em}
     $g' = \dfrac{1}{1 + x^{2}} \cdot 2x$
     
    \begin{align*}
    \int \ln(1 + x^{2})\ dx
     & = x \cdot \ln (1 + x^{2}) - \int x \cdot \dfrac{2x}{1 + x^{2}} \ dx \\\\
     & = x \cdot \ln (1 + x^{2}) - \int \dfrac{2x^{2}}{1 + x^{2}} \ dx \\\\
     & = x \cdot \ln (1 + x^{2}) - 2\cdot\int \dfrac{x^{2}}{1 + x^{2}} \ dx \\\\
     & = x \cdot \ln (1 + x^{2}) - 2\cdot\int \dfrac{x^{2}}{1 + x^{2}} + \dfrac{1 + x^{2}}{1 + x^2} - \dfrac{1 + x^{2}}{1 + x^2}\ dx \\\\
     & = x \cdot \ln (1 + x^{2}) - 2\cdot \left[\int \dfrac{1 + x^{2}}{1 + x^{2}} \ dx - \int \dfrac{1}{1 + x^{2}}\ dx \right] \\\\
     & = x \cdot \ln(1 + x^{2}) - 2 \cdot (x - \arctan x).
    \end{align*}

\paragraph{Ex 11:}
    $\bigint \arctan(x)\ dx$
    
     $\\f = x$, \hspace{1em}
     $f' = 1$, \hspace{1em}
     $g = \arctan x$, \hspace{1em}
     $g' = \dfrac{1}{1+x^{2}}$
    
    \begin{align*}
    \int \arctan(x)\ dx
     & = x \cdot \arctan x - \int \dfrac{x}{x^{2} + 1} \ dx \\\\
     & = x \cdot \arctan x - \int \dfrac{x}{x^{2} + 1} \ dx 
    \end{align*}
     $u = x^{2} + 1$, \hspace{1em}
     $du = 2x \ dx$
     \begin{align*}
     & = x \cdot \arctan x - \int \dfrac{x}{u} \cdot \dfrac{1}{2x} \ du \\\\
     & = x \cdot \arctan x - \dfrac{1}{2} \cdot \int \dfrac{1}{u} \ du \\\\
     & = x \cdot \arctan x - \dfrac{1}{2} \cdot \ln u \\\\
     & = x \cdot \arctan x - \dfrac{1}{2} \cdot \ln (x^{2} + 1).
     \end{align*}

     \begin{align*}

     & = x^{2} \arctan x - x + \arctan x.

\paragraph{Ex 13:}
    $\bigint \sqrt{\tg (x)} \ dx\\\\$
% Resolução 1 -------------------------------------------------------
    Resolução 1:
     $\\u = \sqrt{\tg x}$,\hspace{1em}
     $u^{2} = \tg x$, \hspace{1em}
     $2 \ du = \dfrac{1}{\cos^{2} x} \ dx = (1 + \tg^{2} x) = (1 + u^{4})$, \hspace{1em}
     $dx = \dfrac{2u}{1 + u^{4}} \ du$
    
    \begin{align*}
     \int \sqrt{\tg (x)} \ dx
     & = \int u \cdot \dfrac{2u}{1+u^{4} \ du}\\\\
     & = \int \dfrac{2u^{2}}{1 + u^{4}} 
    \end{align*}
   obs*: 
    $1 + u^{4} = (u^{2} + i)(u^{2} - i)\\\\$ 
    $ \dfrac{2u^{2}}{1 + u^{4}} = \dfrac{1}{(u^{2} + i)} + \dfrac{1}{(u^{2} - i)}\\\\\\$
    $u^{2} + i = (u + i \sqrt{i})(u - i \sqrt{i})$
    $u^{2} - i = (u + \sqrt{i})(u - \sqrt{i})$
    \begin{align*}
     \int \sqrt{\tg (x)} \ dx
     & = \int \dfrac{1}{u^{2} + i} + \int \dfrac{1}{u^{2} - i} \ du \\\\
     & = \int \dfrac{1}{u^{2} + i} \ du + \int \dfrac{1}{u^{2} - i} \ du
    \end{align*}
    
    \begin{align*}
     \int \dfrac{1}{u^{2} + i} \ du
     & = \int \dfrac{1}{(u + i \sqrt{i})(u - i \sqrt{i}) } \ du \\\\
     & = \int \dfrac{1}{2i \sqrt{i}}\left( \dfrac{1}{u - i\sqrt{i}} - \dfrac{1}{u - i\sqrt{i}}\right)\ du \\\\
     & = \dfrac{1}{2i \sqrt{i}} \int\left( \dfrac{1}{u - i\sqrt{i}} - \dfrac{1}{u - i\sqrt{i}}\right)\ du \\\\
     & = \dfrac{1}{2i \sqrt{i}} \left(\int \dfrac{1}{u - i\sqrt{i}} \ du - \int \dfrac{1}{u - i\sqrt{i}} \ du \right) \\\\
     & = \dfrac{1}{2i \sqrt{i}} \left[ \ln(u - i\sqrt{i} - \ln(u + i\sqrt{i})\right]\\\\
     & = \dfrac{1}{2i \sqrt{i}} \cdot \ln \dfrac{u - i\sqrt{i}}{u + i\sqrt{i}}
    \end{align*}
    
    \begin{align*}
     \int \dfrac{1}{u^{2} - i} \ du
     & = \int \dfrac{1}{(u + \sqrt{i})(u - \sqrt{i}) } \ du \\\\
     & = \int \dfrac{1}{2 \sqrt{i}}\left( \dfrac{1}{u - \sqrt{i}} - \dfrac{1}{u + i\sqrt{i}}\right)\ du \\\\
     & = \dfrac{1}{2 \sqrt{i}} \left(\int \dfrac{1}{u - \sqrt{i}} \ du - \int \dfrac{1}{u + i\sqrt{i}} \ du \right) \\\\
     & = \dfrac{1}{2 \sqrt{i}} \left[ \ln(u - \sqrt{i} - \ln(u + \sqrt{i})\right]\\\\
     & = \dfrac{1}{2 \sqrt{i}} \cdot \ln \dfrac{u - \sqrt{i}}{u + \sqrt{i}}
    \end{align*}
    
    \begin{align*}
     \int \dfrac{1}{u^{2} + i} \ du + \int \dfrac{1}{u^{2} - i} \ du
     & = \left(\dfrac{1}{2i \sqrt{i}} \cdot \ln \dfrac{u - i\sqrt{i}}{u + i\sqrt{i}}\right) + \left(\dfrac{1}{2 \sqrt{i}} \cdot \ln \dfrac{u - \sqrt{i}}{u + \sqrt{i}}\right) \\\\
     & = \dfrac{1}{2i \sqrt{i}} \cdot \ln \dfrac{\sqrt{\tg x} - i\sqrt{i}}{\sqrt{\tg x} + i\sqrt{i}} + \dfrac{1}{2 \sqrt{i}} \cdot \ln \dfrac{\sqrt{\tg x} - \sqrt{i}}{\sqrt{\tg x} + \sqrt{i}}.
    \end{align*}
% Resolução 2 -------------------------------------------------------
    Resolução 2:
